184 printf("=== %s ===\n", nome_missione);
185
186 // Obiettivi missione 'Palude'
187 if (tipo_missione == 1)
188 printf("Obiettivo: Sconfiggi %d Generali Orco (%d/%d)\n", target, progress, target);
189
190 // Obiettivi missione 'Magione'
191 else if (tipo_missione == 2)

193
194 // Obiettivi missione 'Grotta'
195 else if (tipo_missione == 3)
196 printf("Obiettivo: Sconfiggi Drago (%d/1) e prendi Spada (%d/1)\n", progress, g->ha_spada_eroe);

201 // Menu azioni
202 if (n_stanze < 10)
203 printf("1. Esplora\n");
204 printf("2. Negozio\n3. Inventario\n4. Torna al Villaggio (Completa Obbiettivo oppure Paga 50 monete)\n\n");
205 printf("Seleziona una delle opzioni del menu [1-4]: ");
206
207 // Gestione scelta utente
208 int scelta = leggi_intero();
209 if (scelta == 1 && n_stanze < 10)
210 {
211 n_stanze++;
212 int dado = lancia_dado(6);

214 // Forza spawn boss se fine dungeon

331 {
332 if (completata)
333 {
334 printf("\nMISSIONE COMPIUTA!\n");
335 getchar();
336 return 1;
337 }
338 if (g->monete >= 50)
339 {
340 g->monete -= 50;
341 return 0;
342 }
343 printf("Non hai 50 monete!\n");
344 getchar();
345 }
346 } while (1);
