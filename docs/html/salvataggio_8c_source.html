44void aggiungi_salvataggio(NodoSalvataggio **testa, Giocatore *g)
45{
46 NodoSalvataggio *nuovo = (NodoSalvataggio *)malloc(sizeof(NodoSalvataggio));
47 if (!nuovo)
48 {
49 printf("Errore di allocazione memoria per il salvataggio.\n");
50 return;
51 }
52
53 // Copia i dati del giocatore
54 nuovo->dati_giocatore = *g;
55 ottieni_data_corrente(nuovo->timestamp, sizeof(nuovo->timestamp));
56 nuovo->prossimo = NULL;
57
58 // Calcolo ID: Massimo attuale + 1
59 int max_id = 0;
60 NodoSalvataggio *curr = *testa;
61 while (curr)
62 {
63 if (curr->id > max_id)
64 max_id = curr->id;
65 curr = curr->prossimo;
66 }
67 nuovo->id = max_id + 1;
68
69 // Inserimento in testa alla lista
70 nuovo->prossimo = *testa;
71 *testa = nuovo;
72
73 // Persistenza su file
74 salva_tutto_su_file(*testa);
75
76 printf("Partita salvata con successo! (ID: %d)\n", nuovo->id);

190 NodoSalvataggio *temp = *testa;

193 temp->prossimo = nuovo;

195 }
